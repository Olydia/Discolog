	\section{Algorithms}

	 	\begin{figure}[]
	 		\begin{algorithmic}[1]\small
	 			\Function{isAcceptable}{$value$, $relation$}
	 			
	 			\If{($relation = dominant$)} 
	 			\State return $(score(value)> bestScoreOfPreference \times 0.7)$
	 			\EndIf
	 			\If{($relation = peer$)} 
	 			\State return $(score(value)> 0)$
	 			\EndIf
	 			\If{($relation = sub$)} 
	 			\State return $(score(value)> 0$  or $isInOAS(value))$
	 			\EndIf
	 			\EndFunction
	 		\end{algorithmic}
	 		\vskip 8pt
	 		\defig{pseudo}{Function to compute the acceptability of a value of preference}
	 	\end{figure} 
	 	
	 	
	 	\begin{figure}[]
	 		\begin{algorithmic}[1]\small
	 			\Function{ComputeAcceptableProposals}{$Value, relation$} \Comment{Value is a criterion type (eg:cuisine,cost) or option(eg: restaurant)}
	 			\State SelfAcceptable($Value$) \Comment{List of acceptable values. see \fig{pseudo}}
	 			\State OtherAcceptable($Value$) \Comment{List of values which are acceptable for the user.}
	 			\State proposals \Comment{ListofProposals}
	 			\If{($relation = dominant$)} 
		 			\ForAll{ $(V: SelfAcceptable)$}
	 	 			\State \textbf{if} $V$ was not proposed in the negotiation \textbf{then} $add(V)$ to proposals.
		 			\EndFor
	 			\EndIf
	 			\If{$relation$ = $peer$ or $submissive$}
	 				\ForAll{ $(V: OtherAcceptable)$}
	 				\State \textbf{if} $V$ was not rejected or accepted in the negotiation and $isAcceptable(V)$ \textbf{then} $add(V)$ to proposals.
	 				\EndFor
	 			\EndIf
	 			\State \textbf{return} proposals.
	 			
	 			\EndFunction
	 		\end{algorithmic}
	 		\vskip 8pt
	 		\defig{pseudo}{Function to compute proposals values}
	 	\end{figure} 
	 	
		 	\begin{figure}[]
		 		\begin{algorithmic}[1]\small
		 			\Function{ComputeProposals}{$relation$}
			 		\State Proposals = $ComputeAcceptableProposals(currentDiscussedCriterion, relation)$
			 		\If{(Proposals is empty)} 
				 		\State Proposals = $ComputeAcceptableProposals(OpenNewCriterion, relation)$ \Comment{Open new criterion which has not been negotiated yet}
				 	\EndIf
				 	\If{(Proposals is empty)}
					 	Proposal = $ComputeAcceptableProposals(Options, relation)$ \Comment{compute acceptables options if all criteria have been already discussed}
		 			\EndIf
		 			\State \textbf{return} proposals.
		 			
		 			\EndFunction
		 			
		 		\end{algorithmic}
		 		\vskip 8pt
		 		\defig{pseudo}{Function to compute proposals in a negotiation.}
		 	\end{figure} 
	 	
	 	 	\begin{figure}[]
	 	 		\begin{algorithmic}[1]\small
	 	 			\Function{reactToUser}{$less$, $more$}
	 	 			
	 	 			\If{($more = MostPrefferedValue()$ and $ (*, more) \notin OAS$)} 
		 	 			\State return $(*, more)$
	 	 			\EndIf
	 	 			\If{($more = LeastPrefferedValue()$ and $ (more, *) \notin OAS$)} 
		 	 			\State return $(more, *) $
	 	 			\EndIf
		 	 		\State The conditions $(2,4)$ are also applied for $less$
	 	 			\State $(less',more') = computePreference(less,more)$ \Comment{calculates a preference from a criterion}
	 	 			\If {$(less',more') \in OAS$}
		 	 			\State $(less1,more1)= reactToCriterion(more)$
		 	 			\If {$(less1,more1)=null$}
			 	 			\State $(less2,more2)= reactToCriterion(less)$ 
			 	 			\If{ $(less2,more2)=null$}
				 	 			\State return $(less',more')$
			 	 			\Else  
				 	 			return $(less2,more2)$
			 	 			\EndIf
		 	 			\EndIf
	 	 			\EndIf
	 	 			\State \textbf{return} $(less', more')$
	 	 			\EndFunction
	 	 			
	 	 		\end{algorithmic}
	 	 		\vskip 8pt
	 	 		\defig{react}{Function to react to a preference (less,more)}
	 	 	\end{figure}
	 	 	
	
		 	\begin{figure}[]
		 		\begin{algorithmic}[1]\small
		 			\Function{ReactTorejectableProposal}{(Proposal)}
						\If{(Proposal isCriterionProposal)}
							\State \textbf{return} reactToCriterion(Proposal)
						\EndIf
						\If{(Proposal isOptionProposal)}
						\State \textbf{return} reactToCriterion(getLeastScoredValue(Proposal))
						\State \Comment{getLeastScoredValue(Proposal) returns the criterion value of the proposal with the least score}
						\EndIf
					
		 			
		 			\EndFunction
		 			
		 		\end{algorithmic}
		 		\vskip 8pt
		 		\defig{pseudo}{Function to compute a preference if the proposal is not acceptable.}
		 		
		 	\end{figure}
		 	\begin{figure}[]
		 		\begin{algorithmic}[1]\small
		 			\Function{CounterPropose}{(Proposal)}
		 			\If{(Proposal isCriterionProposal and isAcceptable(Proposal))}
		 			\State \textbf{return} ComputeAcceptableProposal() \Comment {compute an acceptable option that contains the acceptable criterion}
		 			\EndIf
		 			\If{isNotAccepted(Proposal)}
		 			\State \textbf{return} computeProposal() \Comment{Compute a counter proposal}

		 			\EndIf
		 			
		 			
		 			\EndFunction
		 			
		 		\end{algorithmic}
		 		\vskip 8pt
		 		\defig{pseudo}{Function to compute a counter proposal.}
		\end{figure}

 				\State \textbf{return} true
 				\EndIf
 				\EndFunction
